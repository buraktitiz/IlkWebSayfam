## Install

```bash
npm install --save deagle
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'deagle'
import 'deagle/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
